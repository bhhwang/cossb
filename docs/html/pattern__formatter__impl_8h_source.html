306 // ISO 8601 offset from UTC in timezone (+-HH:MM)
307 class z_formatter :public flag_formatter
308 {
309 public:
310  const std::chrono::seconds cache_refresh = std::chrono::seconds(5);
311 
312  z_formatter() :_last_update(std::chrono::seconds(0)) {}
313  z_formatter(const z_formatter&) = delete;
314  z_formatter& operator=(const z_formatter&) = delete;
315 
316  void format(details::log_msg& msg, const std::tm& tm_time) override
317  {
318 #ifdef _WIN32
319  int total_minutes = get_cached_offset(msg, tm_time);
320 #else
321  // No need to chache under gcc,
322  // it is very fast (already stored in tm.tm_gmtoff)
323  int total_minutes = os::utc_minutes_offset(tm_time);
324 #endif
325 
326  int h = total_minutes / 60;
327  int m = total_minutes % 60;
328  char sign = h >= 0 ? '+' : '-';
329  msg.formatted << sign;
330  pad_n_join(msg.formatted, h, m, ':');
331  }
332 private:
333  log_clock::time_point _last_update;
334  int _offset_minutes;
335  std::mutex _mutex;
336 
337  int get_cached_offset(const log_msg& msg, const std::tm& tm_time)
338  {
339  using namespace std::chrono;
340  std::lock_guard<std::mutex> l(_mutex);
341  if (msg.time - _last_update >= cache_refresh)
342  {
343  _offset_minutes = os::utc_minutes_offset(tm_time);
344  _last_update = msg.time;
345  }
346  return _offset_minutes;
347  }
348 };

459 inline void spdlog::pattern_formatter::compile_pattern(const std::string& pattern)
460 {
461  auto end = pattern.end();
462  std::unique_ptr<details::aggregate_formatter> user_chars;
463  for (auto it = pattern.begin(); it != end; ++it)
464  {
465  if (*it == '%')
466  {
467  if (user_chars) //append user chars found so far
468  _formatters.push_back(std::move(user_chars));
469 
470  if (++it != end)
471  handle_flag(*it);
472  else
473  break;
474  }
475  else // chars not following the % sign should be displayed as is
476  {
477  if (!user_chars)
478  user_chars = std::unique_ptr<details::aggregate_formatter>(new details::aggregate_formatter());
479  user_chars->add_ch(*it);
480  }
481  }
482  if (user_chars) //append raw chars found so far
483  {
484  _formatters.push_back(std::move(user_chars));
485  }
