475 template <typename T, std::size_t SIZE, typename Allocator>
476 void MemoryBuffer<T, SIZE, Allocator>::grow(std::size_t size)
477 {
478  std::size_t new_capacity =
479  (std::max)(size, this->capacity_ + this->capacity_ / 2);
480  T *new_ptr = this->allocate(new_capacity);
481  // The following code doesn't throw, so the raw pointer above doesn't leak.
482  std::copy(this->ptr_,
483  this->ptr_ + this->size_, make_ptr(new_ptr, new_capacity));
484  std::size_t old_capacity = this->capacity_;
485  T *old_ptr = this->ptr_;
486  this->capacity_ = new_capacity;
487  this->ptr_ = new_ptr;
488  // deallocate may throw (at least in principle), but it doesn't matter since
489  // the buffer already uses the new storage and will deallocate it in case
490  // of exception.
491  if (old_ptr != data_)
492  this->deallocate(old_ptr, old_capacity);
493 }

703 {

708  // of for every digit. The idea comes from the talk by Alexandrescu
709  // "Three Optimization Tips for C++". See speed-test for a comparison.
710  if (n < 10) return count;

732 {

1054 // An argument visitor.
1055 // To use ArgVisitor define a subclass that implements some or all of the
1056 // visit methods with the same signatures as the methods in ArgVisitor,
1057 // for example, visit_int(int).
1058 // Specify the subclass name as the Impl template parameter. Then calling
1059 // ArgVisitor::visit for some argument will dispatch to a visit method
1060 // specific to the argument type. For example, if the argument type is
1061 // double then visit_double(double) method of a subclass will be called.
1062 // If the subclass doesn't contain a method with this signature, then
1063 // a corresponding method of ArgVisitor will be called.
1064 //
1065 // Example:
1066 // class MyArgVisitor : public ArgVisitor<MyArgVisitor, void> {
1067 // public:
1068 // void visit_int(int value) { print("{}", value); }
1069 // void visit_double(double value) { print("{}", value ); }
1070 // };
1071 //
1072 // ArgVisitor uses the curiously recurring template pattern:
1073 // http://en.wikipedia.org/wiki/Curiously_recurring_template_pattern
1074 template <typename Impl, typename Result>
1075 class ArgVisitor
1076 {
1077 public:
1078  Result visit_unhandled_arg()
1079  {
1080  return Result();
1081  }
1082 
1083  Result visit_int(int value)
1084  {
1085  return FMT_DISPATCH(visit_any_int(value));
1086  }
1087  Result visit_long_long(LongLong value)
1088  {
1089  return FMT_DISPATCH(visit_any_int(value));
1090  }
1091  Result visit_uint(unsigned value)
1092  {
1093  return FMT_DISPATCH(visit_any_int(value));
1094  }
1095  Result visit_ulong_long(ULongLong value)
1096  {
1097  return FMT_DISPATCH(visit_any_int(value));
1098  }
1099  Result visit_char(int value)
1100  {
1101  return FMT_DISPATCH(visit_any_int(value));
1102  }
1103  template <typename T>
1104  Result visit_any_int(T)
1105  {
1106  return FMT_DISPATCH(visit_unhandled_arg());
1107  }
1108 
1109  Result visit_double(double value)
1110  {
1111  return FMT_DISPATCH(visit_any_double(value));
1112  }
1113  Result visit_long_double(long double value)
1114  {
1115  return FMT_DISPATCH(visit_any_double(value));
1116  }
1117  template <typename T>
1118  Result visit_any_double(T)
1119  {
1120  return FMT_DISPATCH(visit_unhandled_arg());
1121  }
1122 
1123  Result visit_string(Arg::StringValue<char>)
1124  {
1125  return FMT_DISPATCH(visit_unhandled_arg());
1126  }
1127  Result visit_wstring(Arg::StringValue<wchar_t>)
1128  {
1129  return FMT_DISPATCH(visit_unhandled_arg());
1130  }
1131  Result visit_pointer(const void *)
1132  {
1133  return FMT_DISPATCH(visit_unhandled_arg());
1134  }
1135  Result visit_custom(Arg::CustomValue)
1136  {
1137  return FMT_DISPATCH(visit_unhandled_arg());
1138  }
1139 
1140  Result visit(const Arg &arg)
1141  {
1142  switch (arg.type)
1143  {
1144  default:
1145  assert(false);
1146  return Result();
1147  case Arg::INT:
1148  return FMT_DISPATCH(visit_int(arg.int_value));
1149  case Arg::UINT:
1150  return FMT_DISPATCH(visit_uint(arg.uint_value));
1151  case Arg::LONG_LONG:
1152  return FMT_DISPATCH(visit_long_long(arg.long_long_value));
1153  case Arg::ULONG_LONG:
1154  return FMT_DISPATCH(visit_ulong_long(arg.ulong_long_value));
1155  case Arg::DOUBLE:
1156  return FMT_DISPATCH(visit_double(arg.double_value));
1157  case Arg::LONG_DOUBLE:
1158  return FMT_DISPATCH(visit_long_double(arg.long_double_value));
1159  case Arg::CHAR:
1160  return FMT_DISPATCH(visit_char(arg.int_value));
1161  case Arg::CSTRING:
1162  {
1163  Value::StringValue<char> str = arg.string;
1164  str.size = 0;
1165  return FMT_DISPATCH(visit_string(str));
1166  }
1167  case Arg::STRING:
1168  return FMT_DISPATCH(visit_string(arg.string));
1169  case Arg::WSTRING:
1170  return FMT_DISPATCH(visit_wstring(arg.wstring));
1171  case Arg::POINTER:
1172  return FMT_DISPATCH(visit_pointer(arg.pointer));
1173  case Arg::CUSTOM:
1174  return FMT_DISPATCH(visit_custom(arg.custom));
1175  }
1176  }
