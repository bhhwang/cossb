66 template<class Mutex>
67 class rotating_file_sink : public base_sink < Mutex >
68 {
69 public:
70  rotating_file_sink(const std::string &base_filename, const std::string &extension,
71  std::size_t max_size, std::size_t max_files,
72  bool force_flush = false) :
73  _base_filename(base_filename),

75  _max_size(max_size),
76  _max_files(max_files),
77  _current_size(0),

81  }
82 
83 protected:
84  void _sink_it(const details::log_msg& msg) override
85  {
86  _current_size += msg.formatted.size();
87  if (_current_size > _max_size)
88  {
89  _rotate();
90  _current_size = msg.formatted.size();
91  }
92  _file_helper.write(msg);
93  }

95 private:
96  static std::string calc_filename(const std::string& filename, std::size_t index, const std::string& extension)
97  {
98  fmt::MemoryWriter w;
99  if (index)
100  w.write("{}.{}.{}", filename, index, extension);
101  else
102  w.write("{}.{}", filename, extension);
103  return w.str();
104  }

106  // Rotate files:
107  // log.txt -> log.1.txt
108  // log.1.txt -> log2.txt
109  // log.2.txt -> log3.txt
110  // log.3.txt -> delete
111 
112  void _rotate()
113  {
114  _file_helper.close();
115  for (auto i = _max_files; i > 0; --i)
116  {
117  std::string src = calc_filename(_base_filename, i - 1, _extension);
118  std::string target = calc_filename(_base_filename, i, _extension);
119 
120  if (details::file_helper::file_exists(target))
121  {
122  if (std::remove(target.c_str()) != 0)
123  {
124  throw spdlog_ex("rotating_file_sink: failed removing " + target);
125  }
126  }
127  if (details::file_helper::file_exists(src) && std::rename(src.c_str(), target.c_str()))
128  {
129  throw spdlog_ex("rotating_file_sink: failed renaming " + src + " to " + target);
130  }
131  }
132  _file_helper.reopen(true);
133  }
134  std::string _base_filename;
135  std::string _extension;
136  std::size_t _max_size;
137  std::size_t _max_files;
138  std::size_t _current_size;
139  details::file_helper _file_helper;
140 };

146  * Rotating file sink based on date. rotates at midnight
147  */
148 template<class Mutex>
149 class daily_file_sink :public base_sink < Mutex >
150 {
151 public:
152  //create daily file sink which rotates on given time
153  daily_file_sink(
154  const std::string& base_filename,
155  const std::string& extension,
156  int rotation_hour,
157  int rotation_minute,
158  bool force_flush = false) : _base_filename(base_filename),
159  _extension(extension),
160  _rotation_h(rotation_hour),
161  _rotation_m(rotation_minute),
162  _file_helper(force_flush)
163  {
164  if (rotation_hour < 0 || rotation_hour > 23 || rotation_minute < 0 || rotation_minute > 59)
165  throw spdlog_ex("daily_file_sink: Invalid rotation time in ctor");
166  _rotation_tp = _next_rotation_tp();
167  _file_helper.open(calc_filename(_base_filename, _extension));
168  }
169 
170 protected:
171  void _sink_it(const details::log_msg& msg) override
172  {
173  if (std::chrono::system_clock::now() >= _rotation_tp)
174  {
175  _file_helper.open(calc_filename(_base_filename, _extension));
176  _rotation_tp = _next_rotation_tp();
177  }
178  _file_helper.write(msg);
179  }
180 
181 private:
182  std::chrono::system_clock::time_point _next_rotation_tp()
183  {
184  using namespace std::chrono;
185  auto now = system_clock::now();
186  time_t tnow = std::chrono::system_clock::to_time_t(now);
187  tm date = spdlog::details::os::localtime(tnow);
188  date.tm_hour = _rotation_h;
189  date.tm_min = _rotation_m;
190  date.tm_sec = 0;
191  auto rotation_time = std::chrono::system_clock::from_time_t(std::mktime(&date));
192  if (rotation_time > now)
193  return rotation_time;
194  else
195  return system_clock::time_point(rotation_time + hours(24));
196  }
197 
198  //Create filename for the form basename.YYYY-MM-DD.extension
199  static std::string calc_filename(const std::string& basename, const std::string& extension)
200  {
201  std::tm tm = spdlog::details::os::localtime();
202  fmt::MemoryWriter w;
203  w.write("{}_{:04d}-{:02d}-{:02d}_{:02d}-{:02d}.{}", basename, tm.tm_year + 1900, tm.tm_mon + 1, tm.tm_mday, tm.tm_hour, tm.tm_min, extension);
204  return w.str();
205  }
