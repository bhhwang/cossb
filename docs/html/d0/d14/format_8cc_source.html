129 // Portable thread-safe version of strerror.
130 // Sets buffer to point to a string describing the error code.
131 // This can be either a pointer to a string stored in buffer,
132 // or a pointer to some static immutable string.
133 // Returns one of the following values:
134 // 0 - success
135 // ERANGE - buffer is not large enough to store the error message
136 // other - failure
137 // Buffer should be at least of size 1.
138 int safe_strerror(
139  int error_code, char *&buffer, std::size_t buffer_size) FMT_NOEXCEPT{
140  assert(buffer != 0 && buffer_size != 0);
141  int result = 0;
142 #if ((_POSIX_C_SOURCE >= 200112L || _XOPEN_SOURCE >= 600) && !_GNU_SOURCE) || __ANDROID__
143  // XSI-compliant version of strerror_r.
144  result = strerror_r(error_code, buffer, buffer_size);
145  if (result != 0)
146  result = errno;
147 #elif _GNU_SOURCE
148  // GNU-specific version of strerror_r.
149  char *message = strerror_r(error_code, buffer, buffer_size);
150  // If the buffer is full then the message is probably truncated.
151  if (message == buffer && strlen(buffer) == buffer_size - 1)
152  result = ERANGE;
153  buffer = message;
154 #elif __MINGW32__
155  errno = 0;
156  (void)buffer_size;
157  buffer = strerror(error_code);
158  result = errno;
159 #elif _WIN32
160  result = strerror_s(buffer, buffer_size, error_code);
161  // If the buffer is full then the message is probably truncated.
162  if (result == 0 && std::strlen(buffer) == buffer_size - 1)
163  result = ERANGE;
164 #else
165  result = strerror_r(error_code, buffer, buffer_size);
166  if (result == -1)
167  result = errno; // glibc versions before 2.13 return result in errno.
168 #endif
169  return result;
170 }

172 void format_error_code(fmt::Writer &out, int error_code,
173  fmt::StringRef message) FMT_NOEXCEPT{
174  // Report error code making sure that the output fits into
175  // INLINE_BUFFER_SIZE to avoid dynamic memory allocation and potential
176  // bad_alloc.
177  out.clear();
178  static const char SEP[] = ": ";
179  static const char ERR[] = "error ";
180  fmt::internal::IntTraits<int>::MainType ec_value = error_code;
181  // Subtract 2 to account for terminating null characters in SEP and ERR.
182  std::size_t error_code_size =
183  sizeof(SEP) + sizeof(ERR) + fmt::internal::count_digits(ec_value) - 2;
184  if (message.size() <= fmt::internal::INLINE_BUFFER_SIZE - error_code_size)
185  out << message << SEP;
186  out << ERR << error_code;
187  assert(out.size() <= fmt::internal::INLINE_BUFFER_SIZE);
188 }

209 // Parses an unsigned integer advancing s to the end of the parsed input.
210 // This function assumes that the first character of s is a digit.
211 template <typename Char>
212 int parse_nonnegative_int(const Char *&s) {
213  assert('0' <= *s && *s <= '9');
214  unsigned value = 0;
215  do {
216  unsigned new_value = value * 10 + (*s++ - '0');
217  // Check if value wrapped around.
218  if (new_value < value) {

734 template <typename Char>
735 unsigned fmt::internal::PrintfFormatter<Char>::parse_header(
736  const Char *&s, FormatSpec &spec) {
737  unsigned arg_index = UINT_MAX;
738  Char c = *s;
739  if (c >= '0' && c <= '9') {
740  // Parse an argument index (if followed by '$') or a width possibly
741  // preceded with '0' flag(s).
742  unsigned value = parse_nonnegative_int(s);
743  if (*s == '$') { // value is an argument index
744  ++s;
745  arg_index = value;
746  }
747  else {
748  if (c == '0')
749  spec.fill_ = '0';
750  if (value != 0) {
751  // Nonzero value means that we parsed width and don't need to
752  // parse it or flags again, so return now.
753  spec.width_ = value;
754  return arg_index;
755  }
756  }
757  }
758  parse_flags(spec, s);
759  // Parse width.
760  if (*s >= '0' && *s <= '9') {
761  spec.width_ = parse_nonnegative_int(s);
762  }
763  else if (*s == '*') {
764  ++s;
765  spec.width_ = WidthHandler(spec).visit(get_arg(s));
766  }
767  return arg_index;
768 }

770 template <typename Char>
771 void fmt::internal::PrintfFormatter<Char>::format(
772  BasicWriter<Char> &writer, BasicStringRef<Char> format_str,
773  const ArgList &args) {
774  const Char *start = format_str.c_str();
775  set_args(args);
776  const Char *s = start;
777  while (*s) {
778  Char c = *s++;
779  if (c != '%') continue;
780  if (*s == c) {
781  write(writer, start, s);
782  start = ++s;
783  continue;
784  }
785  write(writer, start, s - 1);
786 
787  FormatSpec spec;
788  spec.align_ = ALIGN_RIGHT;
789 
790  // Parse argument index, flags and width.
791  unsigned arg_index = parse_header(s, spec);
792 
793  // Parse precision.
794  if (*s == '.') {
795  ++s;
796  if ('0' <= *s && *s <= '9') {
797  spec.precision_ = parse_nonnegative_int(s);
798  }
799  else if (*s == '*') {
800  ++s;
801  spec.precision_ = PrecisionHandler().visit(get_arg(s));
802  }
803  }
804 
805  Arg arg = get_arg(s, arg_index);
806  if (spec.flag(HASH_FLAG) && IsZeroInt().visit(arg))
807  spec.flags_ &= ~HASH_FLAG;
808  if (spec.fill_ == '0') {
809  if (arg.type <= Arg::LAST_NUMERIC_TYPE)
810  spec.align_ = ALIGN_NUMERIC;
811  else
812  spec.fill_ = ' '; // Ignore '0' flag for non-numeric types.
813  }
814 
815  // Parse length and convert the argument to the required type.
816  switch (*s++) {
817  case 'h':
818  if (*s == 'h')
819  ArgConverter<signed char>(arg, *++s).visit(arg);
820  else
821  ArgConverter<short>(arg, *s).visit(arg);
822  break;
823  case 'l':
824  if (*s == 'l')
825  ArgConverter<fmt::LongLong>(arg, *++s).visit(arg);
826  else
827  ArgConverter<long>(arg, *s).visit(arg);
828  break;
829  case 'j':
830  ArgConverter<intmax_t>(arg, *s).visit(arg);
831  break;
832  case 'z':
833  ArgConverter<size_t>(arg, *s).visit(arg);
834  break;
835  case 't':
836  ArgConverter<ptrdiff_t>(arg, *s).visit(arg);
837  break;
838  case 'L':
839  // printf produces garbage when 'L' is omitted for long double, no
840  // need to do the same.
841  break;
842  default:
843  --s;
844  ArgConverter<int>(arg, *s).visit(arg);
845  }
846 
847  // Parse type.
848  if (!*s)
849  FMT_THROW(FormatError("invalid format string"));
850  spec.type_ = static_cast<char>(*s++);
851  if (arg.type <= Arg::LAST_INTEGER_TYPE) {
852  // Normalize type.
853  switch (spec.type_) {
854  case 'i':
855  case 'u':
856  spec.type_ = 'd';
857  break;
858  case 'c':
859  // TODO: handle wchar_t
860  CharConverter(arg).visit(arg);
861  break;
862  }
863  }
864 
865  start = s;
866 
867  // Format argument.
868  switch (arg.type) {
869  case Arg::INT:
870  writer.write_int(arg.int_value, spec);
871  break;
872  case Arg::UINT:
873  writer.write_int(arg.uint_value, spec);
874  break;
875  case Arg::LONG_LONG:
876  writer.write_int(arg.long_long_value, spec);
877  break;
878  case Arg::ULONG_LONG:
879  writer.write_int(arg.ulong_long_value, spec);
880  break;
881  case Arg::CHAR: {
882  if (spec.type_ && spec.type_ != 'c')
883  writer.write_int(arg.int_value, spec);
884  typedef typename BasicWriter<Char>::CharPtr CharPtr;
885  CharPtr out = CharPtr();
886  if (spec.width_ > 1) {
887  Char fill = ' ';
888  out = writer.grow_buffer(spec.width_);
889  if (spec.align_ != ALIGN_LEFT) {
890  std::fill_n(out, spec.width_ - 1, fill);
891  out += spec.width_ - 1;
892  }
893  else {
894  std::fill_n(out + 1, spec.width_ - 1, fill);
895  }
896  }
897  else {
898  out = writer.grow_buffer(1);
899  }
900  *out = static_cast<Char>(arg.int_value);
901  break;
902  }
903  case Arg::DOUBLE:
904  writer.write_double(arg.double_value, spec);
905  break;
906  case Arg::LONG_DOUBLE:
907  writer.write_double(arg.long_double_value, spec);
908  break;
909  case Arg::CSTRING:
910  arg.string.size = 0;
911  writer.write_str(arg.string, spec);
912  break;
913  case Arg::STRING:
914  writer.write_str(arg.string, spec);
915  break;
916  case Arg::WSTRING:
917  writer.write_str(ignore_incompatible_str<Char>(arg.wstring), spec);
918  break;
919  case Arg::POINTER:
920  if (spec.type_ && spec.type_ != 'p')
921  internal::report_unknown_type(spec.type_, "pointer");
922  spec.flags_ = HASH_FLAG;
923  spec.type_ = 'x';
924  writer.write_int(reinterpret_cast<uintptr_t>(arg.pointer), spec);
925  break;
926  case Arg::CUSTOM: {
927  if (spec.type_)
928  internal::report_unknown_type(spec.type_, "object");
929  const void *str_format = "s";
930  arg.custom.format(&writer, arg.custom.value, &str_format);
931  break;
932  }
933  default:
934  assert(false);
935  break;
936  }
937  }
938  write(writer, start, s);
939 }

941 template <typename Char>
942 const Char *fmt::BasicFormatter<Char>::format(
943  const Char *&format_str, const Arg &arg) {
944  const Char *s = format_str;
945  FormatSpec spec;
946  if (*s == ':') {
947  if (arg.type == Arg::CUSTOM) {
948  arg.custom.format(this, arg.custom.value, &s);
949  return s;
950  }
951  ++s;
952  // Parse fill and alignment.
953  if (Char c = *s) {
954  const Char *p = s + 1;
955  spec.align_ = ALIGN_DEFAULT;
956  do {
957  switch (*p) {
958  case '<':
959  spec.align_ = ALIGN_LEFT;
960  break;
961  case '>':
962  spec.align_ = ALIGN_RIGHT;
963  break;
964  case '=':
965  spec.align_ = ALIGN_NUMERIC;
966  break;
967  case '^':
968  spec.align_ = ALIGN_CENTER;
969  break;
970  }
971  if (spec.align_ != ALIGN_DEFAULT) {
972  if (p != s) {
973  if (c == '}') break;
974  if (c == '{')
975  FMT_THROW(FormatError("invalid fill character '{'"));
976  s += 2;
977  spec.fill_ = c;
978  }
979  else ++s;
980  if (spec.align_ == ALIGN_NUMERIC)
981  require_numeric_argument(arg, '=');
982  break;
983  }
984  } while (--p >= s);
985  }
986 
987  // Parse sign.
988  switch (*s) {
989  case '+':
990  check_sign(s, arg);
991  spec.flags_ |= SIGN_FLAG | PLUS_FLAG;
992  break;
993  case '-':
994  check_sign(s, arg);
995  spec.flags_ |= MINUS_FLAG;
996  break;
997  case ' ':
998  check_sign(s, arg);
999  spec.flags_ |= SIGN_FLAG;
1000  break;
1001  }
1002 
1003  if (*s == '#') {
1004  require_numeric_argument(arg, '#');
1005  spec.flags_ |= HASH_FLAG;
1006  ++s;
1007  }
1008 
1009  // Parse width and zero flag.
1010  if ('0' <= *s && *s <= '9') {
1011  if (*s == '0') {
1012  require_numeric_argument(arg, '0');
1013  spec.align_ = ALIGN_NUMERIC;
1014  spec.fill_ = '0';
1015  }
1016  // Zero may be parsed again as a part of the width, but it is simpler
1017  // and more efficient than checking if the next char is a digit.
1018  spec.width_ = parse_nonnegative_int(s);
1019  }
1020 
1021  // Parse precision.
1022  if (*s == '.') {
1023  ++s;
1024  spec.precision_ = 0;
1025  if ('0' <= *s && *s <= '9') {
1026  spec.precision_ = parse_nonnegative_int(s);
1027  }
1028  else if (*s == '{') {
1029  ++s;
1030  const Arg &precision_arg = parse_arg_index(s);
1031  if (*s++ != '}')
1032  FMT_THROW(FormatError("invalid format string"));
1033  ULongLong value = 0;
1034  switch (precision_arg.type) {
1035  case Arg::INT:
1036  if (precision_arg.int_value < 0)
1037  FMT_THROW(FormatError("negative precision"));
1038  value = precision_arg.int_value;
1039  break;
1040  case Arg::UINT:
1041  value = precision_arg.uint_value;
1042  break;
1043  case Arg::LONG_LONG:
1044  if (precision_arg.long_long_value < 0)
1045  FMT_THROW(FormatError("negative precision"));
1046  value = precision_arg.long_long_value;
1047  break;
1048  case Arg::ULONG_LONG:
1049  value = precision_arg.ulong_long_value;
1050  break;
1051  default:
1052  FMT_THROW(FormatError("precision is not integer"));
1053  }
1054  if (value > INT_MAX)
1055  FMT_THROW(FormatError("number is too big"));
1056  spec.precision_ = static_cast<int>(value);
1057  }
1058  else {
1059  FMT_THROW(FormatError("missing precision specifier"));
1060  }
1061  if (arg.type < Arg::LAST_INTEGER_TYPE || arg.type == Arg::POINTER) {
1062  FMT_THROW(FormatError(
1063  fmt::format("precision not allowed in {} format specifier",
1064  arg.type == Arg::POINTER ? "pointer" : "integer")));
1065  }
1066  }
1067 
1068  // Parse type.
1069  if (*s != '}' && *s)
1070  spec.type_ = static_cast<char>(*s++);
1071  }
1072 
1073  if (*s++ != '}')
1074  FMT_THROW(FormatError("missing '}' in format string"));
1075  start_ = s;
1076 
1077  // Format argument.
1078  internal::ArgFormatter<Char>(*this, spec, s - 1).visit(arg);
1079  return s;
1080 }
